import sys
from PySide6.QtWidgets import (
    QApplication, QWidget, QLabel,
    QLineEdit, QPushButton, QVBoxLayout,
    QMessageBox
)


class LoginWindow(QWidget):
    def __init__(self):
        super().__init__()

        self.setWindowTitle("Bejelentkezés")
        self.setFixedSize(300, 180)

        # Felhasználónév mező
        self.username_label = QLabel("Felhasználónév:")
        self.username_input = QLineEdit()

        # Jelszó mező
        self.password_label = QLabel("Jelszó:")
        self.password_input = QLineEdit()
        self.password_input.setEchoMode(QLineEdit.Password)

        # Bejelentkezés gomb
        self.login_button = QPushButton("Bejelentkezés")
        self.login_button.clicked.connect(self.handle_login)

        # Layout
        layout = QVBoxLayout()
        layout.addWidget(self.username_label)
        layout.addWidget(self.username_input)
        layout.addWidget(self.password_label)
        layout.addWidget(self.password_input)
        layout.addWidget(self.login_button)

        self.setLayout(layout)

    def handle_login(self):
        username = self.username_input.text()
        password = self.password_input.text()

        # Egyszerű teszt ellenőrzés
        if username == "admin" and password == "1234":
            QMessageBox.information(self, "Siker", "Sikeres bejelentkezés!")
        else:
            QMessageBox.warning(self, "Hiba", "Hibás felhasználónév vagy jelszó!")


if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = LoginWindow()
    window.show()
    sys.exit(app.exec())
